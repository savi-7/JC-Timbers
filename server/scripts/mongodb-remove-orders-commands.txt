db.orders.find().count()

// 3. Remove ALL orders (WARNING: This will delete everything)
db.orders.deleteMany({})

// 4. Remove orders by status
// Remove pending orders
db.orders.deleteMany({ status: "pending" })

// Remove completed orders
db.orders.deleteMany({ status: "completed" })

// Remove cancelled orders
db.orders.deleteMany({ status: "cancelled" })

// 5. Remove orders by date range
// Remove orders created before a specific date
db.orders.deleteMany({ createdAt: { $lt: new Date("2024-01-01") } })

// Remove orders created between two dates
db.orders.deleteMany({ 
  createdAt: { 
    $gte: new Date("2024-01-01"), 
    $lte: new Date("2024-12-31") 
  } 
})

// 6. Remove orders for specific users (if needed)
// Replace "USER_ID" with actual user ID
